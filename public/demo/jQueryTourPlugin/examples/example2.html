Example 2

This is a an example of the jTour plugin with two steps and some options

Code

$(document).ready(function() {

	var tourdata = [
	   {
		  html: "Hello World"
	   },{
		  html: "Welcome to the Tour",
		  live: 5000,
		  delayIn: 500
	   }
	];
	
	var myTour = jTour(tourdata);
	myTour.start();

});
